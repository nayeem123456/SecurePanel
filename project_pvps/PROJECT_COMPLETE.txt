STEP 1: UPDATE DATABASE
Run this SQL in phpMyAdmin:

```sql
USE pvps_db;

-- Add columns to users table
ALTER TABLE users 
ADD COLUMN palm_image_path VARCHAR(255) AFTER biometric_template,
ADD COLUMN palm_registered BOOLEAN DEFAULT FALSE AFTER palm_image_path,
ADD COLUMN last_palm_scan TIMESTAMP NULL AFTER palm_registered,
ADD COLUMN updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP;

-- Create palm_scans table
CREATE TABLE palm_scans (
    scan_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    scan_type ENUM('registration', 'authentication', 'transaction'),
    image_path VARCHAR(255) NOT NULL,
    image_hash VARCHAR(64) NOT NULL,
    biometric_signature TEXT NOT NULL,
    confidence_score DECIMAL(5, 4) NOT NULL,
    scan_timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);

-- Create palm_analytics table
CREATE TABLE palm_analytics (
    analytics_id INT AUTO_INCREMENT PRIMARY KEY,
    scan_id INT NOT NULL,
    vein_pattern_score DECIMAL(5, 4),
    palm_lines_score DECIMAL(5, 4),
    skin_texture_score DECIMAL(5, 4),
    finger_geometry_score DECIMAL(5, 4),
    palm_shape_score DECIMAL(5, 4),
    image_quality_score DECIMAL(5, 4),
    overall_confidence DECIMAL(5, 4) NOT NULL,
    analysis_data JSON,
    FOREIGN KEY (scan_id) REFERENCES palm_scans(scan_id) ON DELETE CASCADE
);

-- Create palm_matches table
CREATE TABLE palm_matches (
    match_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    scan_id INT NOT NULL,
    reference_scan_id INT NOT NULL,
    match_score DECIMAL(5, 4) NOT NULL,
    is_match BOOLEAN NOT NULL,
    match_type ENUM('login', 'transaction', 'verification'),
    match_timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);
```
